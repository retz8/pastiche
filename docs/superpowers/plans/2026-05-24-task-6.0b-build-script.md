# Task 6.0b — Build Script + Plugin Packaging Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create a `scripts/build.ts` orchestrator that produces a complete `dist-plugin/` directory — the loadable shape of pastiche as a Claude Code plugin — with a single command (`npx tsx scripts/build.ts`).

**Architecture:** The build script is a single TypeScript file that orchestrates five steps in sequence: (1) clean `dist-plugin/`, (2) render agents via adapter templates, (3) copy skills into plugin directory structure, (4) copy templates, (5) generate `plugin.json`, (6) bundle JS via `tsup`, (7) compile + copy Rust binary. A separate `tsup.config.ts` configures the JS bundler. No runtime dependencies beyond `tsx` (already installed) and `tsup` (new dev dep).

**Tech Stack:** TypeScript (`tsx` for execution), `tsup` (esbuild-based bundler, new dev dep), Rust toolchain (for `cargo build`).

### Task 2: Create tsup.config.ts

**Files:**
- Create: `tsup.config.ts`

- [ ] **Step 1: Create `tsup.config.ts` at repo root**

This config bundles `scripts/extract-fact-ts.ts` → `dist-plugin/dist/extract-fact.js`. The entry point is renamed via the `entry` object syntax so the output filename drops the `-ts` suffix.

```typescript
import { defineConfig } from 'tsup';

export default defineConfig({
  entry: { 'extract-fact': 'scripts/extract-fact-ts.ts' },
  outDir: 'dist-plugin/dist',
  format: ['esm'],
  target: 'node20',
  platform: 'node',
  splitting: false,
  clean: false,
  banner: {
    js: '#!/usr/bin/env node',
  },
});
```

Key decisions:
- `entry` object maps output name `extract-fact` to source file `extract-fact-ts.ts`, producing `extract-fact.js`
- `format: ['esm']` because the repo is `"type": "module"`
- `clean: false` because `build.ts` handles cleaning `dist-plugin/` itself
- `banner` adds shebang so `node dist-plugin/dist/extract-fact.js` works when invoked directly
- `target: 'node20'` — minimum node version for modern features

### Task 3: Create scripts/build.ts — the orchestrator

**Files:**
- Create: `scripts/build.ts`

This is the core of 6.0b. The build script does everything in sequence:

1. Clean `dist-plugin/`
2. Render agents (template + sidecar + canonical body → `dist-plugin/agents/*.md`)
3. Copy skills (`skills/<name>.md` → `dist-plugin/skills/<name>/SKILL.md`)
4. Copy templates (`templates/*` → `dist-plugin/templates/`)
5. Generate `plugin.json` → `dist-plugin/.claude-plugin/plugin.json`
6. Run `tsup` (bundles extractor JS)
7. Run `cargo build --release` + copy binary → `dist-plugin/bin/pastiche-lint`

- [ ] **Step 1: Create `scripts/build.ts`**

```typescript
import * as fs from 'node:fs';
import * as path from 'node:path';
import { execSync } from 'node:child_process';

const ROOT = path.resolve(import.meta.dirname, '..');
const DIST = path.join(ROOT, 'dist-plugin');

function log(msg: string) {
  console.log(`[build] ${msg}`);
}

function clean() {
  if (fs.existsSync(DIST)) {
    fs.rmSync(DIST, { recursive: true });
  }
  log('cleaned dist-plugin/');
}

function ensureDir(dir: string) {
  fs.mkdirSync(dir, { recursive: true });
}

function renderAgents() {
  const agentsDir = path.join(ROOT, 'agents');
  const templatePath = path.join(ROOT, 'adapters', 'claude-code', 'agents.template');
  const template = fs.readFileSync(templatePath, 'utf8');

  const bodyRegex = /^---\n[\s\S]*?\n---\n\n([\s\S]*)$/;
  const templateBody = template.match(bodyRegex);
  if (!templateBody) {
    throw new Error('agents.template: could not extract template body after frontmatter');
  }
  const tmpl = templateBody[1];

  const outDir = path.join(DIST, 'agents');
  ensureDir(outDir);

  const canonicalFiles = fs.readdirSync(agentsDir)
    .filter(f => f.endsWith('.md') && !f.includes('.meta.'));

  for (const file of canonicalFiles) {
    const name = file.replace('.md', '');
    const sidecarPath = path.join(agentsDir, `${name}.claude-code.meta.yaml`);
    if (!fs.existsSync(sidecarPath)) {
      log(`  skip ${name} (no .claude-code.meta.yaml sidecar)`);
      continue;
    }

    const body = fs.readFileSync(path.join(agentsDir, file), 'utf8');
    const sidecar = fs.readFileSync(sidecarPath, 'utf8').trimEnd();

    const rendered = tmpl
      .replace('{{sidecar}}', sidecar)
      .replace('{{body}}', body);

    fs.writeFileSync(path.join(outDir, `${name}.md`), rendered);
    log(`  agent: ${name}.md`);
  }
}

function copySkills() {
  const skillsDir = path.join(ROOT, 'skills');
  const files = fs.readdirSync(skillsDir).filter(f => f.endsWith('.md'));

  for (const file of files) {
    const name = file.replace('.md', '');
    const outDir = path.join(DIST, 'skills', name);
    ensureDir(outDir);
    fs.copyFileSync(
      path.join(skillsDir, file),
      path.join(outDir, 'SKILL.md'),
    );
    log(`  skill: ${name}/SKILL.md`);
  }
}

function copyTemplates() {
  const templatesDir = path.join(ROOT, 'templates');
  const outDir = path.join(DIST, 'templates');
  ensureDir(outDir);

  const files = fs.readdirSync(templatesDir);
  for (const file of files) {
    fs.copyFileSync(
      path.join(templatesDir, file),
      path.join(outDir, file),
    );
    log(`  template: ${file}`);
  }
}

function generatePluginJson() {
  const outDir = path.join(DIST, '.claude-plugin');
  ensureDir(outDir);

  const manifest = {
    name: 'pastiche',
    description: 'Faithful design-system execution for LLM coding agents',
    version: '0.0.0-dev',
  };

  fs.writeFileSync(
    path.join(outDir, 'plugin.json'),
    JSON.stringify(manifest, null, 2) + '\n',
  );
  log('  plugin.json generated');
}

function bundleJs() {
  log('bundling JS (tsup)...');
  execSync('npx tsup --config tsup.config.ts', {
    cwd: ROOT,
    stdio: 'inherit',
  });
  log('  JS bundled → dist-plugin/dist/extract-fact.js');
}

function buildRustBinary() {
  const manifestPath = path.join(ROOT, 'rust', 'pastiche-lint', 'Cargo.toml');
  if (!fs.existsSync(manifestPath)) {
    throw new Error('rust/pastiche-lint/Cargo.toml not found — cannot build pastiche-lint binary');
  }

  log('compiling Rust binary (cargo build --release)...');
  execSync(`cargo build --release --manifest-path ${manifestPath}`, {
    cwd: ROOT,
    stdio: 'inherit',
  });

  const binaryName = process.platform === 'win32' ? 'pastiche-lint.exe' : 'pastiche-lint';
  const src = path.join(ROOT, 'rust', 'pastiche-lint', 'target', 'release', binaryName);
  if (!fs.existsSync(src)) {
    throw new Error(`compiled binary not found at ${src}`);
  }

  const binDir = path.join(DIST, 'bin');
  ensureDir(binDir);
  fs.copyFileSync(src, path.join(binDir, binaryName));
  fs.chmodSync(path.join(binDir, binaryName), 0o755);
  log(`  binary: bin/${binaryName}`);
}

function main() {
  log('=== pastiche plugin build ===');

  clean();

  log('rendering agents...');
  renderAgents();

  log('copying skills...');
  copySkills();

  log('copying templates...');
  copyTemplates();

  log('generating plugin.json...');
  generatePluginJson();

  bundleJs();

  buildRustBinary();

  log('=== build complete ===');
  log(`output: ${DIST}`);
}

main();
```
